This project was made by me to understand the working of the most common microcontroller, Arduino Uno R3. From ATL labs of School to Colleges, I have seen using the UNO R3 for teaching the basics of a microcontroller after teachig about breadboards, LEDs, Resistors, and various Modules. This project might look simple, to simply drive a car but the twist in the project is that I have used JOYSTICK controls and also lights which work by code, not just connected to battery :)
The complicated part of this project is managing the joystick controls and logical lighting system. Hope you enjoy reading this project file and hope wish to make one ^_^

1. Arduino UNO R3
Well, this is the original Arduino UNO R3 but you do not necessarily need to buy an original one if you are a beginner. Check out this on Amazon. You may go with any working third party model. Just check for reviews ;)
I will be using the Original Arduino UNO R3 :)
which looks like this,
No need to be suprised. It is an old one of 2022. Functioning is all same but the looks are different :)


2. Motor Shield for Arduino UNO R3
This is something which you have to buy from third party sources as this is not made by Arduino. You may check out a suitable one on Amazon. I am using this from Quartz Components.


3. HC-05 Bluetooth Module
This is something which you have to buy from third party sources as this is not made by Arduino. You may check out a suitable one on Amazon like this. I am using this from Quartz Components.


4. A suitable chasis
This is something on which you will put all your electronics. I am using one from Amazon.


5. LED Lights
I am using four 8mm warm white color LED, four 5mm yellow LED and two 5mm red LED.


6. Batteries
I am using the most commonly used 18650 Lithium-ion Batteries with a battery holder.


7. Charging Module for Li-ion Batteries
I am not using the commonly used TP4056 Charging Module which is enough to charge a single Li-ion battery like the one I have used, but I will be using two such in series. So, I ordered from Amazon a Step-Up booster charging module that boost the 5V to 8.4V, which is the optimum volatge for charging two li-ion batteries in series. Also, I will be using this BMS from Amazon, to make sure that both the batteries are charged at their maximum voltage which is 8.4V. The BMS I am using is of 20 amperes. You may definitely choose a lower ampere one. It is recommended to add a BMS for longer life of batteries.


8. Passive Buzzer with 3 pin
A silent car is too boring!! Don't you agree? :)
For various events, I am using this passive buzzer that can generate tone from 1.5 ~ 2.5 kHz.

> coding()

This code also contains parts of code that also print output in serial monitor when the Arduino is connected to laptop and Serial Monitor of Arduino IDE is opened. You can use this code to ensure that your HC-05 module is connected perfectly synced with Arduino. You may also connect LEDs to check whether the LEDS are working or not. You may also motor shield to check whether the motor shield is working or not by seeing the indicator light.
Do not forget to remove the Power Jumper from motor shield. It is small cap like (usually yellow in color) often near the external power screw terminals. This damage your Arduino.
Do not connect the Arduino to laptop when the motors are connected to the motor shield. This may damage your laptop's USB port as motors may draw a large amount of current which the USB port might not be able to handle.

bt_rc_car_debug.ino
#include <AFMotor.h> 
#include <SoftwareSerial.h> 

// --- BLUETOOTH SETUP ---
// RX is A3 (Connects to HC-05 TX)
// TX is A4 (Connects to HC-05 RX)
SoftwareSerial BT(A3, A4); 

// --- PIN DEFINITIONS & 4WD MOTORS ---
AF_DCMotor motorLeftFront(1);  // M1 block
AF_DCMotor motorLeftRear(2);   // M2 block
AF_DCMotor motorRightFront(3); // M3 block
AF_DCMotor motorRightRear(4);  // M4 block

#define HL_PIN 9       // Headlight (PWM - Servo 1 Pin)
#define TL_PIN 10      // Taillight (PWM - Servo 2 Pin)
#define IND_L A0       // Indicator Left 
#define IND_R A1       // Indicator Right 
#define BUZZER_PIN A2  // Passive Buzzer 
#define BT_STATE_PIN 2 // HC-05 State Pin 
#define STATUS_LED 13  // Uno Built-in LED

// --- STATE VARIABLES ---
bool headlightU = false;
bool taillightV = false;
bool parkingLightOn = false;
bool hazardLightOn = false;
bool indicatorL = false;
bool indicatorR = false;
bool wasConnected = false;

unsigned long hornTimer = 0;
unsigned long specialZTimer = 0;
unsigned long parkingLockBeepEnd = 0;

void setup() {
Serial.begin(9600); 
BT.begin(9600);     

Serial.println("=================================");
Serial.println("  4WD RC CAR SYSTEM BOOTING...   ");
Serial.println("=================================");

// Initialize all 4 Motors to stopped
motorLeftFront.setSpeed(0); motorLeftFront.run(RELEASE);
motorLeftRear.setSpeed(0);  motorLeftRear.run(RELEASE);
motorRightFront.setSpeed(0); motorRightFront.run(RELEASE);
motorRightRear.setSpeed(0);  motorRightRear.run(RELEASE);

pinMode(HL_PIN, OUTPUT); pinMode(TL_PIN, OUTPUT);
pinMode(IND_L, OUTPUT); pinMode(IND_R, OUTPUT);
pinMode(BUZZER_PIN, OUTPUT);
pinMode(STATUS_LED, OUTPUT);
pinMode(BT_STATE_PIN, INPUT);

// Startup Beep
tone(BUZZER_PIN, 1500, 200); delay(250); 
tone(BUZZER_PIN, 2000, 300); delay(300);

Serial.println("[SYSTEM] Ready! Waiting for Bluetooth connection...");
}

void loop() {
readBluetooth();
updateLights();
updateBuzzer();
updateStatusLed();
}

// --- BLUETOOTH PARSING ---
void readBluetooth() {
if (BT.available()) {
    String cmd = BT.readStringUntil('\n');
    cmd.trim(); 
    
    if (cmd.length() == 0) return;

    Serial.print("[BT RX] Raw Command Received: '");
    Serial.print(cmd);
    Serial.println("'");

    // Dynamic Joystick Parsing
    char dirY = cmd[0];
    if (dirY == 'F' || dirY == 'B') {
    int xIndex = cmd.indexOf('L');
    if (xIndex == -1) xIndex = cmd.indexOf('R');

    if (xIndex > 0) { 
        if (parkingLightOn) {
        Serial.println("[WARNING] Movement ignored: Parking Brake is ON!");
        parkingLockBeepEnd = millis() + 3000;
        stopMotors();
        return;
        }

        int spdY = cmd.substring(1, xIndex).toInt();
        char dirX = cmd[xIndex];
        int spdX = cmd.substring(xIndex + 1).toInt();

        int y = constrain(spdY, 0, 255);
        int x = constrain(spdX, 0, 255);
        
        if (dirY == 'B') y = -y;
        if (dirX == 'L') x = -x;

        int leftSpeed = y + x;
        int rightSpeed = y - x;
        
        Serial.print("   -> Motor Cmd | Left Side: ");
        Serial.print(leftSpeed);
        Serial.print(" | Right Side: ");
        Serial.println(rightSpeed);

        setMotors(leftSpeed, rightSpeed);
        return; 
    }
    }

    // Single Character Commands
    char c = cmd[0];
    switch (c) {
    case 'F': case 'B': case 'L': case 'R':
        if (parkingLightOn) {
        Serial.println("[WARNING] Movement ignored: Parking Brake is ON!");
        parkingLockBeepEnd = millis() + 3000;
        stopMotors();
        break;
        }
        if (c == 'F') { setMotors(255, 255); Serial.println("   -> Driving: FORWARD"); }
        if (c == 'B') { setMotors(-255, -255); Serial.println("   -> Driving: BACKWARD"); }
        if (c == 'L') { setMotors(-255, 255); Serial.println("   -> Driving: LEFT"); }
        if (c == 'R') { setMotors(255, -255); Serial.println("   -> Driving: RIGHT"); }
        break;
    case 'S': stopMotors(); Serial.println("   -> Status: STOPPED"); break;
    case 'U': headlightU = true; Serial.println("   -> Lights: Headlights ON"); break;
    case 'u': headlightU = false; Serial.println("   -> Lights: Headlights OFF"); break;
    case 'V': taillightV = true; Serial.println("   -> Lights: Taillights ON"); break;
    case 'v': taillightV = false; Serial.println("   -> Lights: Taillights OFF"); break;
    case 'W': parkingLightOn = true; stopMotors(); Serial.println("   -> State: PARKING BRAKE ENGAGED"); break;
    case 'w': parkingLightOn = false; Serial.println("   -> State: PARKING BRAKE RELEASED"); break;
    case 'X': hazardLightOn = true; indicatorL = false; indicatorR = false; Serial.println("   -> Lights: HAZARDS ON"); break;
    case 'x': hazardLightOn = false; Serial.println("   -> Lights: HAZARDS OFF"); break;
    case '1': indicatorL = true; Serial.println("   -> Lights: Left Indicator ON"); break;
    case '2': indicatorL = false; Serial.println("   -> Lights: Left Indicator OFF"); break;
    case '4': indicatorR = true; Serial.println("   -> Lights: Right Indicator ON"); break;
    case '3': indicatorR = false; Serial.println("   -> Lights: Right Indicator OFF"); break;
    case 'Y': hornTimer = millis(); Serial.println("   -> Action: HORN"); break;       
    case 'Z': specialZTimer = millis(); Serial.println("   -> Action: SPECIAL HIGH-BEAM FLASH"); break;   
    default: Serial.println("   -> [ERROR] Unknown Command!"); break;
    }
}
}

// --- LIGHTING LOGIC ---
void updateLights() {
int hl = 0; 
int tl = 0; 
bool blinkState = (millis() % 1000) < 500; 
bool indLeftOut = false;
bool indRightOut = false;

if (headlightU) { hl = 230; tl = 50; }
if (taillightV) { hl = 230; tl = 255; }

if (parkingLightOn) {
    hl = 128; tl = 128; 
    if (blinkState) { indLeftOut = true; indRightOut = true; }
}

if (hazardLightOn) {
    unsigned long t = millis() % 1900; 
    if (t < 100 || (t >= 200 && t < 300) || (t >= 400 && t < 500)) { hl = 255; } 
    else { hl = 0; }
    if (blinkState) { indLeftOut = true; indRightOut = true; }
} else {
    if (indicatorL && blinkState) indLeftOut = true;
    if (indicatorR && blinkState) indRightOut = true;
}

if (millis() - specialZTimer < 300) { hl = 255; }

analogWrite(HL_PIN, hl);
analogWrite(TL_PIN, tl);
digitalWrite(IND_L, indLeftOut ? HIGH : LOW);
digitalWrite(IND_R, indRightOut ? HIGH : LOW);
}

// --- BUZZER LOGIC ---
void updateBuzzer() {
bool isConnected = digitalRead(BT_STATE_PIN);

if (isConnected && !wasConnected) {
    Serial.println("\n[SYSTEM] *** BLUETOOTH DEVICE CONNECTED! ***\n");
    tone(BUZZER_PIN, 1800, 150); delay(150);
    tone(BUZZER_PIN, 2500, 250); delay(250);
} else if (!isConnected && wasConnected) {
    Serial.println("\n[SYSTEM] *** BLUETOOTH DEVICE DISCONNECTED! ***\n");
}
wasConnected = isConnected;

if (millis() - hornTimer < 300) {
    tone(BUZZER_PIN, 3000); 
} else if (millis() < parkingLockBeepEnd) {
    tone(BUZZER_PIN, 800);
} else if (!isConnected) {
    if (millis() % 1000 < 100) { tone(BUZZER_PIN, 1000); } 
    else { noTone(BUZZER_PIN); }
} else {
    noTone(BUZZER_PIN);
}
}

// --- STATUS LED LOGIC ---
void updateStatusLed() {
unsigned long t = millis() % 2500;
if (t < 100 || (t >= 200 && t < 300) || (t >= 400 && t < 500)) { digitalWrite(STATUS_LED, HIGH); } 
else { digitalWrite(STATUS_LED, LOW); }
}

// --- 4WD SHIELD MOTOR CONTROL ---
void setMotors(int left, int right) {
left = constrain(left, -255, 255);
right = constrain(right, -255, 255);

// --- DRIVE LEFT SIDE (M1 & M2) ---
if (left > 0) { 
    motorLeftFront.setSpeed(left); motorLeftFront.run(FORWARD); 
    motorLeftRear.setSpeed(left);  motorLeftRear.run(FORWARD);
} 
else if (left < 0) { 
    motorLeftFront.setSpeed(-left); motorLeftFront.run(BACKWARD); 
    motorLeftRear.setSpeed(-left);  motorLeftRear.run(BACKWARD);
} 
else { 
    motorLeftFront.setSpeed(0); motorLeftFront.run(RELEASE); 
    motorLeftRear.setSpeed(0);  motorLeftRear.run(RELEASE);
}

// --- DRIVE RIGHT SIDE (M3 & M4) ---
if (right > 0) { 
    motorRightFront.setSpeed(right); motorRightFront.run(FORWARD); 
    motorRightRear.setSpeed(right);  motorRightRear.run(FORWARD);
} 
else if (right < 0) { 
    motorRightFront.setSpeed(-right); motorRightFront.run(BACKWARD); 
    motorRightRear.setSpeed(-right);  motorRightRear.run(BACKWARD);
} 
else { 
    motorRightFront.setSpeed(0); motorRightFront.run(RELEASE); 
    motorRightRear.setSpeed(0);  motorRightRear.run(RELEASE);
}
}

void stopMotors() {
motorLeftFront.setSpeed(0); motorLeftFront.run(RELEASE);
motorLeftRear.setSpeed(0);  motorLeftRear.run(RELEASE);
motorRightFront.setSpeed(0); motorRightFront.run(RELEASE);
motorRightRear.setSpeed(0);  motorRightRear.run(RELEASE);
}
